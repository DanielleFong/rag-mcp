Expand description

Fluent builder for constructing elicitation schemas.

ยงExample

use rmcp::model::*;

let schema = ElicitationSchema::builder()
    .required_email("email")
    .required_integer("age", 0, 150)
    .optional_bool("newsletter", false)
    .description("User registration")
    .build();
