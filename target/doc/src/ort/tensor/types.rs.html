4/// Enum mapping ONNX Runtime's supported tensor data types.
5#[derive(Debug, PartialEq, Eq, Clone, Copy)]
6pub enum TensorElementType {
7	/// 32-bit floating point number, equivalent to Rust's `f32`.
8	Float32,
9	/// Unsigned 8-bit integer, equivalent to Rust's `u8`.
10	Uint8,
11	/// Signed 8-bit integer, equivalent to Rust's `i8`.
12	Int8,
13	/// Unsigned 16-bit integer, equivalent to Rust's `u16`.
14	Uint16,
15	/// Signed 16-bit integer, equivalent to Rust's `i16`.
16	Int16,
17	/// Signed 32-bit integer, equivalent to Rust's `i32`.
18	Int32,
19	/// Signed 64-bit integer, equivalent to Rust's `i64`.
20	Int64,
21	/// String, equivalent to Rust's `String`.
22	String,
23	/// Boolean, equivalent to Rust's `bool`.
24	Bool,
25	/// 16-bit floating point number, equivalent to [`half::f16`] (with the `half` feature).
26	Float16,
27	/// 64-bit floating point number, equivalent to Rust's `f64`. Also known as `double`.
28	Float64,
29	/// Unsigned 32-bit integer, equivalent to Rust's `u32`.
30	Uint32,
31	/// Unsigned 64-bit integer, equivalent to Rust's `u64`.
32	Uint64,
33	/// Brain 16-bit floating point number, equivalent to [`half::bf16`] (with the `half` feature).
34	Bfloat16,
35	Complex64,
36	Complex128,
37	/// 8-bit floating point number with 4 exponent bits and 3 mantissa bits, with only NaN values and no infinite
38	/// values.
39	Float8E4M3FN,
40	/// 8-bit floating point number with 4 exponent bits and 3 mantissa bits, with only NaN values, no infinite
41	/// values, and no negative zero.
42	Float8E4M3FNUZ,
43	/// 8-bit floating point number with 5 exponent bits and 2 mantissa bits.
44	Float8E5M2,
45	/// 8-bit floating point number with 5 exponent bits and 2 mantissa bits, with only NaN values, no infinite
46	/// values, and no negative zero.
47	Float8E5M2FNUZ,
48	/// 4-bit unsigned integer.
49	Uint4,
50	/// 4-bit signed integer.
51	Int4,
52	Undefined
53}

55impl TensorElementType {
56	/// Returns the size in bytes that a container of this type occupies according to its total capacity.
57	pub fn byte_size(&self, container_capacity: usize) -> usize {
58		match self {
59			TensorElementType::Uint4 | TensorElementType::Int4 => container_capacity / 2,
60			TensorElementType::Bool | TensorElementType::Int8 | TensorElementType::Uint8 => container_capacity,
61			TensorElementType::Int16 | TensorElementType::Uint16 => container_capacity * 2,
62			TensorElementType::Int32 | TensorElementType::Uint32 => container_capacity * 4,
63			TensorElementType::Int64 | TensorElementType::Uint64 => container_capacity * 8,
64			TensorElementType::String => 0, // unsure what to do about this...
65			TensorElementType::Float8E4M3FN | TensorElementType::Float8E4M3FNUZ | TensorElementType::Float8E5M2 | TensorElementType::Float8E5M2FNUZ => {
66				container_capacity * 4
67			}
68			TensorElementType::Float16 | TensorElementType::Bfloat16 => container_capacity * 2,
69			TensorElementType::Float32 => container_capacity * 4,
70			TensorElementType::Float64 => container_capacity * 8,
71			TensorElementType::Complex64 => container_capacity * 8,
72			TensorElementType::Complex128 => container_capacity * 16,
73			TensorElementType::Undefined => 0
74		}
75	}
76}

168/// Trait used to map Rust types (for example `f32`) to ONNX tensor element data types (for example `Float`).
169pub trait IntoTensorElementType {
170	/// Returns the ONNX tensor element data type corresponding to the given Rust type.
171	fn into_tensor_element_type() -> TensorElementType;
172
173	private_trait!();
174}
175
176/// A superset of [`IntoTensorElementType`] that represents traits whose underlying memory is identical between Rust and
177/// C++ (i.e., every type except `String`).
178pub trait PrimitiveTensorElementType: IntoTensorElementType {
179	private_trait!();
180}
181
182macro_rules! impl_type_trait {
183	($type_:ty, $variant:ident) => {
184		impl IntoTensorElementType for $type_ {
185			fn into_tensor_element_type() -> TensorElementType {
186				TensorElementType::$variant
187			}
188
189			private_impl!();
190		}
191
192		impl PrimitiveTensorElementType for $type_ {
193			private_impl!();
194		}
195	};
196}

230/// Adapter for common Rust string types to ONNX strings.
231pub trait Utf8Data {
232	/// Returns the contents of this value as a slice of UTF-8 bytes.
233	fn as_utf8_bytes(&self) -> &[u8];
234}

236impl Utf8Data for String {

238		self.as_bytes()
239	}
