6/// [MIGraphX execution provider](https://onnxruntime.ai/docs/execution-providers/MIGraphX-ExecutionProvider.html) for
7/// hardware acceleration with AMD GPUs.
8#[derive(Debug, Default, Clone)]
9pub struct MIGraphX {
10	device_id: i32,
11	enable_fp16: bool,
12	enable_int8: bool,
13	use_native_calibration_table: bool,
14	int8_calibration_table_name: Option<CString>,
15	save_model_path: Option<CString>,
16	load_model_path: Option<CString>,
17	exhaustive_tune: bool
18}
19
20super::impl_ep!(MIGraphX);
21
22impl MIGraphX {
23	/// Configures which device the EP should use.
24	///
25	/// ```
26	/// # use ort::{ep, session::Session};
27	/// # fn main() -> ort::Result<()> {
28	/// let ep = ep::MIGraphX::default().with_device_id(0).build();
29	/// # Ok(())
30	/// # }
31	/// ```
32	#[must_use]
33	pub fn with_device_id(mut self, device_id: i32) -> Self {
34		self.device_id = device_id;
35		self
36	}
37
38	/// Enable FP16 quantization for the model.
39	///
40	/// ```
41	/// # use ort::{ep, session::Session};
42	/// # fn main() -> ort::Result<()> {
43	/// let ep = ep::MIGraphX::default().with_fp16(true).build();
44	/// # Ok(())
45	/// # }
46	/// ```
47	#[must_use]
48	pub fn with_fp16(mut self, enable: bool) -> Self {
49		self.enable_fp16 = enable;
50		self
51	}
52
53	/// Enable 8-bit integer quantization for the model. Requires
54	/// [`MIGraphX::with_int8_calibration_table`] to be set.
55	///
56	/// ```
57	/// # use ort::{ep, session::Session};
58	/// # fn main() -> ort::Result<()> {
59	/// let ep = ep::MIGraphX::default().with_int8(true).with_int8_calibration_table("...", false).build();
60	/// # Ok(())
61	/// # }
62	/// ```
63	#[must_use]
64	pub fn with_int8(mut self, enable: bool) -> Self {
65		self.enable_int8 = enable;
66		self
67	}
68
69	/// Configures the path to the input calibration data for int8 quantization.
70	///
71	/// The `native` parameter specifies the format the calibration data is in - `true` for native int8 format, `false`
72	/// for the JSON dump format.
73	///
74	/// ```
75	/// # use ort::{ep, session::Session};
76	/// # fn main() -> ort::Result<()> {
77	/// let ep = ep::MIGraphX::default().with_int8(true).with_int8_calibration_table("...", false).build();
78	/// # Ok(())
79	/// # }
80	/// ```
81	#[must_use]
82	pub fn with_int8_calibration_table(mut self, table_name: impl AsRef<str>, native: bool) -> Self {
83		self.use_native_calibration_table = native;
84		self.int8_calibration_table_name = Some(CString::new(table_name.as_ref()).expect("invalid string"));
85		self
86	}
87
88	/// Save the compiled MIGraphX model to the given path.
89	///
90	/// The compiled model can then be loaded in subsequent runs with [`MIGraphX::with_load_model`].
91	///
92	/// ```
93	/// # use ort::{ep, session::Session};
94	/// # fn main() -> ort::Result<()> {
95	/// let ep = ep::MIGraphX::default().with_save_model("./compiled_model.mxr").build();
96	/// # Ok(())
97	/// # }
98	/// ```
99	#[must_use]
100	pub fn with_save_model(mut self, path: impl AsRef<str>) -> Self {
101		self.save_model_path = Some(CString::new(path.as_ref()).expect("invalid string"));
102		self
103	}
104
105	/// Load the compiled MIGraphX model (previously generated by [`MIGraphX::with_save_model`]) from
106	/// the given path.
107	///
108	/// ```
109	/// # use ort::{ep, session::Session};
110	/// # fn main() -> ort::Result<()> {
111	/// let ep = ep::MIGraphX::default().with_load_model("./compiled_model.mxr").build();
112	/// # Ok(())
113	/// # }
114	/// ```
115	#[must_use]
116	pub fn with_load_model(mut self, path: impl AsRef<str>) -> Self {
117		self.load_model_path = Some(CString::new(path.as_ref()).expect("invalid string"));
118		self
119	}
120
121	/// Enable exhaustive tuning; trades loading time for inference performance.
122	///
123	/// ```
124	/// # use ort::{ep, session::Session};
125	/// # fn main() -> ort::Result<()> {
126	/// let ep = ep::MIGraphX::default().with_exhaustive_tune(true).build();
127	/// # Ok(())
128	/// # }
129	/// ```
130	#[must_use]
131	pub fn with_exhaustive_tune(mut self, enable: bool) -> Self {
132		self.exhaustive_tune = enable;
133		self
134	}
135}

137impl ExecutionProvider for MIGraphX {
138	fn name(&self) -> &'static str {
139		"MIGraphXExecutionProvider"
140	}
141
142	fn supported_by_platform(&self) -> bool {
143		cfg!(any(all(target_os = "linux", target_arch = "x86_64"), all(target_os = "windows", target_arch = "x86_64")))
144	}
145
146	#[allow(unused, unreachable_code)]
147	fn register(&self, session_builder: &mut SessionBuilder) -> Result<(), RegisterError> {
148		#[cfg(any(feature = "load-dynamic", feature = "migraphx"))]
149		{
150			use core::ptr;
151
152			use crate::{AsPointer, ortsys};
153
154			let options = ort_sys::OrtMIGraphXProviderOptions {
155				device_id: self.device_id,
156				migraphx_fp16_enable: self.enable_fp16.into(),
157				migraphx_int8_enable: self.enable_int8.into(),
158				migraphx_use_native_calibration_table: self.use_native_calibration_table.into(),
159				migraphx_int8_calibration_table_name: self.int8_calibration_table_name.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
160				migraphx_load_compiled_model: self.load_model_path.is_some().into(),
161				migraphx_load_model_path: self.load_model_path.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
162				migraphx_save_compiled_model: self.save_model_path.is_some().into(),
163				migraphx_save_model_path: self.save_model_path.as_ref().map(|c| c.as_ptr()).unwrap_or_else(ptr::null),
164				migraphx_exhaustive_tune: self.exhaustive_tune
165			};
166			ortsys![unsafe SessionOptionsAppendExecutionProvider_MIGraphX(session_builder.ptr_mut(), &options)?];
167			return Ok(());
168		}
169
170		Err(RegisterError::MissingFeature)
171	}
