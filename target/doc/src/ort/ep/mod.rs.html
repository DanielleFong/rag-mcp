1//! [`ExecutionProvider`]s provide hardware acceleration to [`Session`](crate::session::Session)s.
2//!
3//! Sessions can be configured with execution providers via [`SessionBuilder::with_execution_providers`]:
4//!
5//! ```no_run
6//! use ort::{ep, session::Session};
7//!
8//! fn main() -> ort::Result<()> {
9//! 	let session = Session::builder()?
10//! 		.with_execution_providers([ep::CUDA::default().build()])?
11//! 		.commit_from_file("model.onnx")?;
12//!
13//! 	Ok(())
14//! }
15//! ```

81pub trait ExecutionProvider: Any + Send + Sync {
82	/// Returns the identifier of this execution provider used internally by ONNX Runtime.
83	///
84	/// This is the same as what's used in ONNX Runtime's Python API to register this execution provider, i.e.
85	/// [`TVM`]'s identifier is `TvmExecutionProvider`.
86	fn name(&self) -> &'static str;
87
88	/// Returns whether this execution provider is supported on this platform.
89	///
90	/// For example, the CoreML execution provider implements this as:
91	/// ```ignore
92	/// impl ExecutionProvider for CoreML {
93	/// 	fn supported_by_platform() -> bool {
94	/// 		cfg!(target_vendor = "apple")
95	/// 	}
96	/// }
97	/// ```
98	fn supported_by_platform(&self) -> bool {
99		true
100	}
101
102	/// Returns `Ok(true)` if ONNX Runtime was *compiled with support* for this execution provider, and `Ok(false)`
103	/// otherwise.
104	///
105	/// An `Err` may be returned if a serious internal error occurs, in which case your application should probably
106	/// just abort.
107	///
108	/// **Note that this does not always mean the execution provider is *usable* for a specific session.** A model may
109	/// use operators not supported by an execution provider, or the EP may encounter an error while attempting to load
110	/// dependencies during session creation. In most cases (i.e. showing the user an error message if CUDA could not be
111	/// enabled), you'll instead want to manually register this EP via [`ExecutionProvider::register`] and detect
112	/// and handle any errors returned by that function.
113	fn is_available(&self) -> Result<bool> {
114		is_ep_available(self.name())
115	}
116
117	/// Attempts to register this execution provider on the given session.
118	fn register(&self, session_builder: &mut SessionBuilder) -> Result<(), RegisterError>;
119}
120
121/// Trait used for execution providers that can have arbitrary configuration keys applied.
122///
123/// Most execution providers have a small set of configuration options which don't change between ONNX Runtime releases;
124/// others, like the CUDA execution provider, often have options added that go undocumented and thus unimplemented by
125/// `ort`. This allows you to configure these options regardless.
126pub trait ArbitrarilyConfigurableExecutionProvider {
127	fn with_arbitrary_config(self, key: impl ToString, value: impl ToString) -> Self;
128}

140/// Dynamic execution provider container, used to provide a list of multiple types of execution providers when
141/// configuring execution providers for a [`SessionBuilder`] or
142/// [`EnvironmentBuilder`](crate::environment::EnvironmentBuilder).
143///
144/// See [`ExecutionProvider`] for more info on execution providers.
145#[derive(Clone)]
146pub struct ExecutionProviderDispatch {
147	pub(crate) inner: Arc<dyn ExecutionProvider>,
148	error_on_failure: bool
149}
150
151impl ExecutionProviderDispatch {
152	pub(crate) fn new<E: ExecutionProvider + 'static>(ep: E) -> Self {
153		ExecutionProviderDispatch {
154			inner: Arc::new(ep) as _,
155			error_on_failure: false
156		}
157	}
158
159	/// Configures this execution provider to silently log an error if registration of the EP fails.
160	/// This is the default behavior; it can be overridden with [`ExecutionProviderDispatch::error_on_failure`].
161	pub fn fail_silently(mut self) -> Self {
162		self.error_on_failure = false;
163		self
164	}
165
166	/// Configures this execution provider to return an error upon EP registration if registration of this EP fails.
167	/// The default behavior is to silently fail and fall back to the next execution provider, or the CPU provider if no
168	/// registrations succeed.
169	pub fn error_on_failure(mut self) -> Self {
170		self.error_on_failure = true;
171		self
172	}
173
174	/// Attempt to downcast this execution provider to a concrete type `E`.
175	pub fn downcast_ref<E: ExecutionProvider>(&self) -> Option<&E> {
176		<dyn Any>::downcast_ref(&*self.inner)
177	}
178}
179
180impl Debug for ExecutionProviderDispatch {
181	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
182		f.debug_struct(self.inner.name())
183			.field("error_on_failure", &self.error_on_failure)
184			.finish()
185	}
186}

208impl ExecutionProviderOptions {
209	pub fn set(&mut self, key: impl Into<Vec<u8>>, value: impl Into<Vec<u8>>) {
210		self.0
211			.insert(CString::new(key).expect("unexpected nul in key string"), CString::new(value).expect("unexpected nul in value string"));
212	}
213
214	#[allow(unused)]
215	pub fn to_ffi(&self) -> ExecutionProviderOptionsFFI {
216		let (key_ptrs, value_ptrs) = self.0.iter().map(|(k, v)| (k.as_ptr(), v.as_ptr())).unzip();
217		ExecutionProviderOptionsFFI { key_ptrs, value_ptrs }
218	}
219}

242#[derive(Debug)]
243pub enum RegisterError {
244	Error(crate::Error),
245	MissingFeature
246}
247
248impl From<crate::Error> for RegisterError {
249	fn from(value: crate::Error) -> Self {
250		Self::Error(value)
251	}
252}
253
254impl From<RegisterError> for crate::Error {
255	fn from(value: RegisterError) -> Self {
256		match value {
257			RegisterError::Error(e) => e,
258			RegisterError::MissingFeature => {
259				crate::Error::new("The execution provider could not be registered because its corresponding Cargo feature is not enabled.")
260			}
261		}
262	}
263}
264
265impl fmt::Display for RegisterError {
266	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
267		match self {
268			Self::Error(e) => fmt::Display::fmt(e, f),
269			Self::MissingFeature => f.write_str("The execution provider could not be registered because its corresponding Cargo feature is not enabled.")
270		}
271	}
272}

276#[allow(unused)]
277macro_rules! define_ep_register {
278	($symbol:ident($($id:ident: $type:ty),*) -> $rt:ty) => {
279		#[cfg(feature = "load-dynamic")]
280		#[allow(non_snake_case)]
281		let $symbol = unsafe {
282			let dylib = $crate::G_ORT_LIB.get().expect("dylib not yet initialized");
283			let symbol: ::core::result::Result<
284				::libloading::Symbol<unsafe extern "C" fn($($id: $type),*) -> $rt>,
285				::libloading::Error
286			> = dylib.get(stringify!($symbol).as_bytes());
287			match symbol {
288				Ok(symbol) => symbol.into_raw(),
289				Err(e) => {
290					return ::core::result::Result::Err($crate::Error::new(::alloc::format!("Error attempting to load symbol `{}` from dynamic library: {}", stringify!($symbol), e)))?;
291				}
292			}
293		};
294		#[cfg(not(feature = "load-dynamic"))]
295		unsafe extern "C" {
296			fn $symbol($($id: $type),*) -> $rt;
297		}
298	};
299}

303macro_rules! impl_ep {
304	(arbitrary; $symbol:ident) => {
305		$crate::ep::impl_ep!($symbol);
306
307		impl $crate::ep::ArbitrarilyConfigurableExecutionProvider for $symbol {
308			fn with_arbitrary_config(mut self, key: impl ::alloc::string::ToString, value: impl ::alloc::string::ToString) -> Self {
309				self.options.set(key.to_string(), value.to_string());
310				self
311			}
312		}
313	};
314	($symbol:ident) => {
315		impl $symbol {
316			#[must_use]
317			pub fn build(self) -> $crate::ep::ExecutionProviderDispatch {
318				self.into()
319			}
320		}
321
322		impl From<$symbol> for $crate::ep::ExecutionProviderDispatch {
323			fn from(value: $symbol) -> Self {
324				$crate::ep::ExecutionProviderDispatch::new(value)
325			}
326		}
327	};
328}

331pub(crate) fn apply_execution_providers(session_builder: &mut SessionBuilder, eps: &[ExecutionProviderDispatch], source: &'static str) -> Result<()> {
332	fn register_inner(session_builder: &mut SessionBuilder, ep: &ExecutionProviderDispatch, #[allow(unused)] source: &'static str) -> Result<bool> {
333		if let Err(e) = ep.inner.register(session_builder) {
334			if ep.error_on_failure {
335				return Err(e)?;
336			}
337
338			if matches!(e, RegisterError::MissingFeature) {
339				if ep.inner.supported_by_platform() {
340					crate::warn!(%source, "{e}");
341				} else {
342					crate::debug!(%source, "{e} (note: additionally, `{}` may not be supported on this platform)", ep.inner.name());
343				}
344			} else {
345				crate::error!(%source, "An error occurred when attempting to register `{}`: {e}", ep.inner.name());
346			}
347			Ok(false)
348		} else {
349			crate::info!(%source, "Successfully registered `{}`", ep.inner.name());
350			Ok(true)
351		}
352	}
353
354	let mut fallback_to_cpu = !eps.is_empty();
355	for ep in eps {
356		if register_inner(session_builder, ep, source)? {
357			fallback_to_cpu = false;
358		}
359	}
360	if fallback_to_cpu {
361		crate::warn!("No execution providers from {source} registered successfully; may fall back to CPU.");
362	}
363	Ok(())
364}

366fn is_ep_available(name: &str) -> Result<bool> {
367	let mut providers: *mut *mut c_char = ptr::null_mut();
368	let mut num_providers = 0;
369	ortsys![unsafe GetAvailableProviders(&mut providers, &mut num_providers)?];
370	if providers.is_null() {
371		return Ok(false);
372	}
373
374	let _guard = run_on_drop(|| ortsys![unsafe ReleaseAvailableProviders(providers, num_providers).expect("infallible")]);
375
376	for i in 0..num_providers {
377		let avail = match char_p_to_string(unsafe { *providers.offset(i as isize) }) {
378			Ok(avail) => avail,
379			Err(e) => {
380				return Err(e);
381			}
382		};
383		if name == avail {
384			return Ok(true);
385		}
386	}
387
388	Ok(false)
389}
