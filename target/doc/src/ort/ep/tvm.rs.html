6#[derive(Debug, Clone, Copy, PartialEq, Eq)]
7pub enum ExecutorType {
8	GraphExecutor,
9	VirtualMachine
10}
11
12#[derive(Debug, Clone, Copy, PartialEq, Eq)]
13pub enum TuningType {
14	AutoTVM,
15	Ansor
16}
17
18#[derive(Debug, Default, Clone)]
19pub struct TVM {
20	/// Executor type used by TVM. There is a choice between two types, `GraphExecutor` and `VirtualMachine`. Default is
21	/// [`ExecutorType::VirtualMachine`].
22	pub executor: Option<ExecutorType>,
23	/// Path to folder with set of files (`.ro-`, `.so`/`.dll`-files and weights) obtained after model tuning.
24	pub so_folder: Option<String>,
25	/// Whether or not to perform a hash check on the model obtained in the `so_folder`.
26	pub check_hash: Option<bool>,
27	/// A path to a file that contains the pre-computed hash for the ONNX model located in the `so_folder` for checking
28	/// when `check_hash` is `Some(true)`.
29	pub hash_file_path: Option<String>,
30	pub target: Option<String>,
31	pub target_host: Option<String>,
32	pub opt_level: Option<usize>,
33	/// Whether or not all model weights are kept on compilation stage, otherwise they are downloaded on each inference.
34	/// `true` is recommended for best performance and is the default.
35	pub freeze_weights: Option<bool>,
36	pub to_nhwc: Option<bool>,
37	pub tuning_type: Option<TuningType>,
38	/// Path to AutoTVM or Ansor tuning file which gives specifications for given model and target for the best
39	/// performance.
40	pub tuning_file_path: Option<String>,
41	pub input_names: Option<String>,
42	pub input_shapes: Option<String>
43}

47impl ExecutionProvider for TVM {
48	fn name(&self) -> &'static str {
49		"TvmExecutionProvider"
50	}
51
52	#[allow(unused, unreachable_code)]
53	fn register(&self, session_builder: &mut SessionBuilder) -> Result<(), RegisterError> {
54		#[cfg(any(feature = "load-dynamic", feature = "tvm"))]
55		{
56			use alloc::format;
57
58			use crate::AsPointer;
59
60			super::define_ep_register!(OrtSessionOptionsAppendExecutionProvider_Tvm(options: *mut ort_sys::OrtSessionOptions, opt_str: *const core::ffi::c_char) -> ort_sys::OrtStatusPtr);
61			let mut option_string = Vec::new();
62			if let Some(check_hash) = self.check_hash {
63				option_string.push(format!("check_hash:{}", if check_hash { "True" } else { "False" }));
64			}
65			if let Some(executor) = self.executor {
66				option_string.push(format!(
67					"executor:{}",
68					match executor {
69						ExecutorType::GraphExecutor => "graph",
70						ExecutorType::VirtualMachine => "vm"
71					}
72				));
73			}
74			if let Some(freeze_weights) = self.freeze_weights {
75				option_string.push(format!("freeze_weights:{}", if freeze_weights { "True" } else { "False" }));
76			}
77			if let Some(hash_file_path) = self.hash_file_path.as_ref() {
78				option_string.push(format!("hash_file_path:{hash_file_path}"));
79			}
80			if let Some(input_names) = self.input_names.as_ref() {
81				option_string.push(format!("input_names:{input_names}"));
82			}
83			if let Some(input_shapes) = self.input_shapes.as_ref() {
84				option_string.push(format!("input_shapes:{input_shapes}"));
85			}
86			if let Some(opt_level) = self.opt_level {
87				option_string.push(format!("opt_level:{opt_level}"));
88			}
89			if let Some(so_folder) = self.so_folder.as_ref() {
90				option_string.push(format!("so_folder:{so_folder}"));
91			}
92			if let Some(target) = self.target.as_ref() {
93				option_string.push(format!("target:{target}"));
94			}
95			if let Some(target_host) = self.target_host.as_ref() {
96				option_string.push(format!("target_host:{target_host}"));
97			}
98			if let Some(to_nhwc) = self.to_nhwc {
99				option_string.push(format!("to_nhwc:{}", if to_nhwc { "True" } else { "False" }));
100			}
101			let options_string = alloc::ffi::CString::new(option_string.join(",")).expect("invalid option string");
102			return Ok(unsafe {
103				crate::error::status_to_result(OrtSessionOptionsAppendExecutionProvider_Tvm(session_builder.ptr_mut(), options_string.as_ptr()))
104			}?);
105		}
106
107		Err(RegisterError::MissingFeature)
108	}
109}
