13/// Type alias for the `Result` type returned by `ort` functions.
14pub type Result<T, E = Error> = core::result::Result<T, E>;

16pub(crate) trait IntoStatus {
17	fn into_status(self) -> ort_sys::OrtStatusPtr;
18}
19
20impl<T> IntoStatus for Result<T, Error> {
21	fn into_status(self) -> ort_sys::OrtStatusPtr {
22		let (code, message) = match &self {
23			Ok(_) => return ort_sys::OrtStatusPtr(ptr::null_mut()),
24			Err(e) => (ort_sys::OrtErrorCode::ORT_FAIL, e.to_string())
25		};
26		with_cstr(message.as_bytes(), &|message| Ok(ortsys![unsafe CreateStatus(code, message.as_ptr())])).expect("invalid error message")
27	}
28}

30/// An error returned by any `ort` API.
31#[derive(Debug)]
32pub struct Error {
33	code: ErrorCode,
34	msg: String
35}

37impl Error {
38	/// Wrap a custom, user-provided error in an [`ort::Error`](Error)..
39	///
40	/// This can be used to return custom errors from e.g. training dataloaders or custom operators if a non-`ort`
41	/// related operation fails.
42	pub fn wrap<T: CoreError + Send + Sync + 'static>(err: T) -> Self {
43		Error {
44			code: ErrorCode::GenericFailure,
45			msg: err.to_string()
46		}
47	}
48
49	/// Creates a custom [`Error`] with the given message.
50	pub fn new(msg: impl Into<String>) -> Self {
51		Error {
52			code: ErrorCode::GenericFailure,
53			msg: msg.into()
54		}
55	}
56
57	/// Creates a custom [`Error`] with the given [`ErrorCode`] and message.
58	pub fn new_with_code(code: ErrorCode, msg: impl Into<String>) -> Self {
59		Error { code, msg: msg.into() }
60	}
61
62	pub fn code(&self) -> ErrorCode {
63		self.code
64	}
65
66	pub fn message(&self) -> &str {

68	}

141impl From<ort_sys::OrtErrorCode> for ErrorCode {
142	fn from(value: ort_sys::OrtErrorCode) -> Self {
143		match value {
144			ort_sys::OrtErrorCode::ORT_OK => Self::Ok,
145			ort_sys::OrtErrorCode::ORT_FAIL => Self::GenericFailure,
146			ort_sys::OrtErrorCode::ORT_INVALID_ARGUMENT => Self::InvalidArgument,
147			ort_sys::OrtErrorCode::ORT_NO_SUCHFILE => Self::NoSuchFile,
148			ort_sys::OrtErrorCode::ORT_NO_MODEL => Self::NoModel,
149			ort_sys::OrtErrorCode::ORT_ENGINE_ERROR => Self::EngineError,
150			ort_sys::OrtErrorCode::ORT_RUNTIME_EXCEPTION => Self::RuntimeException,
151			ort_sys::OrtErrorCode::ORT_INVALID_PROTOBUF => Self::InvalidProtobuf,
152			ort_sys::OrtErrorCode::ORT_MODEL_LOADED => Self::ModelLoaded,
153			ort_sys::OrtErrorCode::ORT_NOT_IMPLEMENTED => Self::NotImplemented,
154			ort_sys::OrtErrorCode::ORT_INVALID_GRAPH => Self::InvalidGraph,
155			ort_sys::OrtErrorCode::ORT_EP_FAIL => Self::ExecutionProviderFailure,
156			#[allow(unreachable_patterns)]
157			_ => Self::GenericFailure
158		}
159	}

181/// Converts an [`ort_sys::OrtStatusPtr`] to a [`Result`].
182///
183/// **Note that this frees `status`!**
184///
185/// # Safety
186/// The value contained in `status` must be a valid [`ort_sys::OrtStatus`] pointer, or a null pointer (in which case the
187/// result will be `Ok`).
188#[inline]
189pub unsafe fn status_to_result(status: ort_sys::OrtStatusPtr) -> Result<(), Error> {
190	let status = status.0;
191	if status.is_null() {
192		Ok(())
193	} else {
194		#[cold]
195		fn status_to_error(status: *mut ort_sys::OrtStatus) -> Error {
196			let code = ErrorCode::from(ortsys![unsafe GetErrorCode(status)]);
197			let raw: *const c_char = ortsys![unsafe GetErrorMessage(status)];
198			match char_p_to_string(raw) {
199				Ok(msg) => {
200					ortsys![unsafe ReleaseStatus(status)];
201					Error { code, msg }
202				}
203				Err(err) => {
204					ortsys![unsafe ReleaseStatus(status)];
205					Error {
206						code,
207						msg: format!("(failed to convert UTF-8: {err})")
208					}
209				}
210			}
211		}
212
213		Err(status_to_error(status))
214	}
