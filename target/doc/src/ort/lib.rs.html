74/// The minor version of ONNX Runtime used by this version of `ort`.
75///
76/// This is equal to the version of the ONNX Runtime API used by `ort` (available in [`ort-sys`][ort_sys] as
77/// [`ORT_API_VERSION`](ort_sys::ORT_API_VERSION)). `ort` requires that the backend supports *at least* this version of
78/// the API.
79pub const MINOR_VERSION: u32 = ort_sys::ORT_API_VERSION;

84#[cfg(feature = "load-dynamic")]
85pub(crate) fn load_dylib_from_path(path: &std::path::Path) -> Result<bool> {
86	let mut inserter = Some(|| -> crate::Result<libloading::Library> {
87		use core::cmp::Ordering;
88
89		let absolute_path = if path.is_absolute() {
90			path.to_path_buf()
91		} else {
92			let relative = std::env::current_exe()
93				.expect("could not get current executable path")
94				.parent()
95				.expect("executable is root?")
96				.join(path);
97			if relative.exists() { relative } else { path.to_path_buf() }
98		};
99		let lib =
100			unsafe { libloading::Library::new(&absolute_path) }.map_err(|e| Error::new(format!("failed to load from `{}`: {e}", absolute_path.display())))?;
101
102		let base_getter: libloading::Symbol<unsafe extern "C" fn() -> *const ort_sys::OrtApiBase> =
103			unsafe { lib.get(b"OrtGetApiBase") }.map_err(|_| Error::new("expected `OrtGetApiBase` to be present in libonnxruntime"))?;
104		let base: *const ort_sys::OrtApiBase = unsafe { base_getter() };
105		assert!(!base.is_null());
106
107		let version_string = unsafe { ((*base).GetVersionString)() };
108		let version_string = unsafe { CStr::from_ptr(version_string) }.to_string_lossy();
109
110		let lib_minor_version = version_string.split('.').nth(1).map_or(0, |x| x.parse::<u32>().unwrap_or(0));
111		match lib_minor_version.cmp(&MINOR_VERSION) {
112			Ordering::Less => {
113				return Err(Error::new(format!(
114					"ort {} is not compatible with the ONNX Runtime binary found at `{}`; expected version >= '1.{MINOR_VERSION}.x', but got '{version_string}'",
115					env!("CARGO_PKG_VERSION"),
116					absolute_path.display()
117				)));
118			}
119			Ordering::Greater => crate::info!(
120				"ort {} was designed for ONNX Runtime '1.{MINOR_VERSION}.x' and may have compatibility issues with the ONNX Runtime binary found at `{}`, which is version '{version_string}'",
121				env!("CARGO_PKG_VERSION"),
122				absolute_path.display()
123			),
124			Ordering::Equal => {}
125		};
126
127		crate::info!("Loaded ONNX Runtime dylib from \"{}\"; version '{version_string}'", absolute_path.display());
128
129		Ok(lib)
130	});
131	G_ORT_LIB.get_or_try_init(|| (unsafe { inserter.take().unwrap_unchecked() })())?;
132	Ok(inserter.is_none())
133}

146struct ApiPointer(NonNull<ort_sys::OrtApi>);
147unsafe impl Send for ApiPointer {}
148unsafe impl Sync for ApiPointer {}
149
150static G_ORT_API: OnceLock<ApiPointer> = OnceLock::new();
151
152/// Returns a reference to the global [`ort_sys::OrtApi`] object.
153///
154/// # Panics
155/// May panic if:
156/// - The `alternative-backend` feature is enabled and [`set_api`] was not yet called.
157/// - Getting the `OrtApi` struct fails, due to `ort` loading an unsupported version of ONNX Runtime.
158/// - Loading the ONNX Runtime dynamic library fails if the `load-dynamic` feature is enabled.
159#[inline]
160pub fn api() -> &'static ort_sys::OrtApi {
161	#[cfg(feature = "alternative-backend")]
162	let ptr = G_ORT_API
163		.get()
164		.expect(
165			"attempted to use `ort` APIs before initializing a backend\nwhen the `alternative-backend` feature is enabled, `ort::set_api` must be called first to configure the `OrtApi` used by the library"
166		)
167		.0;
168	#[cfg(not(feature = "alternative-backend"))]
169	let ptr = G_ORT_API.get_or_init(setup_api).0;
170	unsafe { ptr.as_ref() }
171}
172
173#[cfg(not(feature = "alternative-backend"))]
174#[cold]
175fn setup_api() -> ApiPointer {
176	#[cfg(feature = "load-dynamic")]
177	let base = unsafe {
178		let dylib = if let Some(handle) = G_ORT_LIB.get() {
179			handle
180		} else {
181			let path: std::path::PathBuf = match std::env::var("ORT_DYLIB_PATH") {
182				Ok(s) if !s.is_empty() => s,
183				#[cfg(target_os = "windows")]
184				_ => "onnxruntime.dll".to_owned(),
185				#[cfg(any(target_os = "linux", target_os = "android"))]
186				_ => "libonnxruntime.so".to_owned(),
187				#[cfg(any(target_os = "macos", target_os = "ios"))]
188				_ => "libonnxruntime.dylib".to_owned()
189			}
190			.into();
191			load_dylib_from_path(&path).expect("Failed to load ONNX Runtime dylib");
192			G_ORT_LIB.get_unchecked()
193		};
194		let base_getter: libloading::Symbol<unsafe extern "C" fn() -> *const ort_sys::OrtApiBase> = dylib
195			.get(b"OrtGetApiBase")
196			.expect("`OrtGetApiBase` must be present in ONNX Runtime dylib");
197		base_getter()
198	};
199	#[cfg(not(feature = "load-dynamic"))]
200	let base = unsafe { ort_sys::OrtGetApiBase() };
201
202	assert!(!base.is_null());
203	let api: *const ort_sys::OrtApi = unsafe { ((*base).GetApi)(ort_sys::ORT_API_VERSION) };
204	ApiPointer(NonNull::new(api.cast_mut()).expect("Failed to initialize ORT API"))
205}

207/// Sets the global [`ort_sys::OrtApi`] interface used by `ort` in order to use alternative backends, or a custom
208/// loading scheme.
209///
210/// When using an alternative backend, this must be called before using any other `ort` API.
211///
212/// Returns `true` if successful, meaning this API will take effect. If an API was already set prior, it will not be
213/// overridden, and this function will return `false`.
214///
215/// ```no_run
216/// # use ort_sys::stub as ort_tract;
217/// ort::set_api(ort_tract::api());
218/// ```
219pub fn set_api(api: ort_sys::OrtApi) -> bool {
220	G_ORT_API.try_insert_with(|| ApiPointer(unsafe { NonNull::new_unchecked(Box::leak(Box::new(api))) }))
221}

223/// Trait to access raw pointers from safe types which wrap unsafe [`ort_sys`] types.
224pub trait AsPointer {
225	/// This safe type's corresponding [`ort_sys`] type.
226	type Sys;
227
228	/// Returns the underlying [`ort_sys`] type pointer this safe type wraps. The pointer is guaranteed to be non-null.
229	fn ptr(&self) -> *const Self::Sys;
230
231	/// Returns the underlying [`ort_sys`] type pointer this safe type wraps as a mutable pointer. The pointer is
232	/// guaranteed to be non-null.
233	fn ptr_mut(&mut self) -> *mut Self::Sys {
234		self.ptr().cast_mut()
235	}
236}

238#[macro_export]
239macro_rules! ortsys {
240	($method:ident) => {
241		($crate::api().$method)
242	};
243	(unsafe $method:ident($($n:expr),* $(,)?)) => {
244		ortsys![@ort: unsafe $method($($n),*)]
245	};
246	(unsafe $method:ident($($n:expr),* $(,)?).expect($e:expr)) => {
247		ortsys![@ort: unsafe $method($($n),*) as Result].expect($e)
248	};
249	(unsafe $method:ident($($n:expr),* $(,)?).expect($e:expr); nonNull($($check:ident),+ $(,)?)$(;)?) => {
250		ortsys![unsafe $method($($n),*).expect($e)];
251		ortsys![@nonNull_panic; $($check),+];
252	};
253	(unsafe $method:ident($($n:expr),* $(,)?); nonNull($($check:ident),+ $(,)?)$(;)?) => {
254		let _x = ortsys![unsafe $method($($n),*)];
255		ortsys![@nonNull_panic; $($check),+];
256		_x
257	};
258	(unsafe $method:ident($($n:expr),* $(,)?)?) => {
259		ortsys![@ort: unsafe $method($($n),+) as Result]?;
260	};
261	(unsafe $method:ident($($n:expr),* $(,)?)?; nonNull($($check:ident),+)$(;)?) => {
262		ortsys![unsafe $method($($n),*)?];
263		ortsys![@nonNull?; $($check),+];
264	};
265
266	(@nonNull_panic; $($check:ident),+) => {
267		$(
268			let Some($check) = $crate::__private::core::ptr::NonNull::new($check as *mut _) else {
269				$crate::util::cold();
270				$crate::__private::core::panic!(concat!("expected `", stringify!($check), "` to not be null"));
271			};
272		)+
273	};
274	(@nonNull?; $($check:ident),+) => {
275		$(
276			let Some($check) = $crate::__private::core::ptr::NonNull::new($check as *mut _) else {
277				$crate::util::cold();
278				return Err($crate::Error::new(concat!("expected `", stringify!($check), "` to not be null")));
279			};
280		)+
281	};
282
283	(@ort: unsafe $method:ident($($n:expr),*)) => {
284		unsafe { ($crate::api().$method)($($n),*) }
285	};
286	(@ort: unsafe $method:ident($($n:expr),*) as Result) => {
287		unsafe { $crate::error::status_to_result(($crate::api().$method)($($n),+)) }
288	};
289	(@$api:ident: unsafe $method:ident($($n:expr),*)) => {
290		unsafe { ($crate::api::$api().unwrap().$method)($($n),+) }
291	};
292	(@$api:ident: unsafe $method:ident($($n:expr),*)?) => {
293		$crate::api::$api().and_then(|api| unsafe { $crate::error::status_to_result((api.$method)($($n),+)) })?
294	};
295	(@$api:ident: unsafe $method:ident($($n:expr),*)?; nonNull($($check:ident),+)$(;)?) => {
296		$crate::api::$api().and_then(|api| unsafe { $crate::error::status_to_result((api.$method)($($n),+)) })?;
297		ortsys![@nonNull?; $($check),+];
298	};
299	(@$api:ident: unsafe $method:ident($($n:expr),*) as Result) => {
300		$crate::api::$api().and_then(|api| unsafe { $crate::error::status_to_result((api.$method)($($n),+)) })
301	};
