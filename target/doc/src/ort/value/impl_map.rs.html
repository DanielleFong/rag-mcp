87impl<Type: MapValueTypeMarker + ?Sized> Value<Type> {
88	pub fn try_extract_key_values<K: IntoTensorElementType + Clone + Hash + Eq, V: PrimitiveTensorElementType + Clone>(&self) -> Result<Vec<(K, V)>> {
89		match self.dtype() {
90			ValueType::Map { key, value } => {
91				let k_type = K::into_tensor_element_type();
92				if k_type != *key {
93					return Err(Error::new_with_code(ErrorCode::InvalidArgument, format!("Cannot extract Map<{:?}, _> (value has K type {:?})", k_type, key)));
94				}
95				let v_type = V::into_tensor_element_type();
96				if v_type != *value {
97					return Err(Error::new_with_code(
98						ErrorCode::InvalidArgument,
99						format!("Cannot extract Map<{}, {}> from Map<{}, {}>", K::into_tensor_element_type(), V::into_tensor_element_type(), k_type, v_type)
100					));
101				}
102
103				let allocator = Allocator::default();
104
105				let mut key_tensor_ptr = ptr::null_mut();
106				ortsys![unsafe GetValue(self.ptr(), 0, allocator.ptr().cast_mut(), &mut key_tensor_ptr)?; nonNull(key_tensor_ptr)];
107				let key_value: DynTensor = unsafe { Value::from_ptr(key_tensor_ptr, None) };
108				if K::into_tensor_element_type() != TensorElementType::String {
109					let dtype = key_value.dtype();
110					let (key_tensor_shape, key_tensor) = match dtype {
111						ValueType::Tensor { ty, shape, .. } => {
112							let mem = key_value.memory_info();
113							if !mem.is_cpu_accessible() {
114								return Err(Error::new(format!(
115									"Cannot extract from value on device `{}`, which is not CPU accessible",
116									mem.allocation_device().as_str()
117								)));
118							}
119
120							if *ty == K::into_tensor_element_type() {
121								let mut output_array_ptr: *mut K = ptr::null_mut();
122								let output_array_ptr_ptr: *mut *mut K = &mut output_array_ptr;
123								let output_array_ptr_ptr_void: *mut *mut c_void = output_array_ptr_ptr.cast();
124								ortsys![unsafe GetTensorMutableData(key_tensor_ptr.as_ptr(), output_array_ptr_ptr_void)?];
125								if output_array_ptr.is_null() {
126									output_array_ptr = NonNull::dangling().as_ptr();
127								}
128
129								(shape, unsafe { slice::from_raw_parts(output_array_ptr, shape.num_elements()) })
130							} else {
131								return Err(Error::new_with_code(
132									ErrorCode::InvalidArgument,
133									format!(
134										"Cannot extract Map<{}, {}> from Map<{}, {}>",
135										K::into_tensor_element_type(),
136										V::into_tensor_element_type(),
137										k_type,
138										v_type
139									)
140								));
141							}
142						}
143						_ => unreachable!()
144					};
145
146					let mut value_tensor_ptr = ptr::null_mut();
147					ortsys![unsafe GetValue(self.ptr(), 1, allocator.ptr().cast_mut(), &mut value_tensor_ptr)?; nonNull(value_tensor_ptr)];
148					let value_value: DynTensor = unsafe { Value::from_ptr(value_tensor_ptr, None) };
149					let (value_tensor_shape, value_tensor) = value_value.try_extract_tensor::<V>()?;
150
151					assert_eq!(key_tensor_shape.len(), 1);
152					assert_eq!(value_tensor_shape.len(), 1);
153					assert_eq!(key_tensor_shape[0], value_tensor_shape[0]);
154
155					let mut vec = Vec::with_capacity(key_tensor_shape[0] as _);
156					for i in 0..key_tensor_shape[0] as usize {
157						vec.push((key_tensor[i].clone(), value_tensor[i].clone()));
158					}
159					Ok(vec)
160				} else {
161					let (key_tensor_shape, key_tensor) = key_value.try_extract_strings()?;
162					// SAFETY: `IntoTensorElementType` is a private trait, and we only map the `String` type to `TensorElementType::String`,
163					// so at this point, `K` is **always** the `String` type, and this transmute really does nothing but please the type
164					// checker.
165					let key_tensor: Vec<K> = unsafe { mem::transmute(key_tensor) };
166
167					let mut value_tensor_ptr = ptr::null_mut();
168					ortsys![unsafe GetValue(self.ptr(), 1, allocator.ptr().cast_mut(), &mut value_tensor_ptr)?; nonNull(value_tensor_ptr)];
169					let value_value: DynTensor = unsafe { Value::from_ptr(value_tensor_ptr, None) };
170					let (value_tensor_shape, value_tensor) = value_value.try_extract_tensor::<V>()?;
171
172					assert_eq!(key_tensor_shape.len(), 1);
173					assert_eq!(value_tensor_shape.len(), 1);
174					assert_eq!(key_tensor_shape[0], value_tensor_shape[0]);
175
176					let mut vec = Vec::with_capacity(key_tensor_shape[0] as _);
177					for i in 0..key_tensor_shape[0] as usize {
178						vec.push((key_tensor[i].clone(), value_tensor[i].clone()));
179					}
180					Ok(vec.into_iter().collect())
181				}
182			}
183			t => Err(Error::new_with_code(
184				ErrorCode::InvalidArgument,
185				format!("Cannot extract Map<{}, {}> from {t}", K::into_tensor_element_type(), V::into_tensor_element_type())
186			))
187		}
188	}

198	/// Creates a [`Map`] from an iterable emitting `K` and `V`.
199	///
200	/// ```
201	/// # use std::collections::HashMap;
202	/// # use ort::value::Map;
203	/// # fn main() -> ort::Result<()> {
204	/// let mut map = HashMap::<i64, f32>::new();
205	/// map.insert(0, 1.0);
206	/// map.insert(1, 2.0);
207	/// map.insert(2, 3.0);
208	///
209	/// let value = Map::<i64, f32>::new(map)?;
210	///
211	/// assert_eq!(*value.extract_map().get(&0).unwrap(), 1.0);
212	/// # 	Ok(())
213	/// # }
214	/// ```
215	pub fn new(data: impl IntoIterator<Item = (K, V)>) -> Result<Self> {
216		let (keys, values): (Vec<K>, Vec<V>) = data.into_iter().unzip();
217		Self::new_kv(Tensor::from_array((vec![keys.len()], keys))?, Tensor::from_array((vec![values.len()], values))?)
218	}
219}

245impl<K: IntoTensorElementType + Debug + Clone + Hash + Eq + 'static, V: IntoTensorElementType + Debug + Clone + 'static> Value<MapValueType<K, V>> {
246	/// Creates a [`Map`] from two tensors of keys & values respectively.
247	///
248	/// ```
249	/// # use std::collections::HashMap;
250	/// # use ort::value::{Map, Tensor};
251	/// # fn main() -> ort::Result<()> {
252	/// let keys = Tensor::<i64>::from_array(([4], vec![0, 1, 2, 3]))?;
253	/// let values = Tensor::<f32>::from_array(([4], vec![1., 2., 3., 4.]))?;
254	///
255	/// let value = Map::new_kv(keys, values)?;
256	///
257	/// assert_eq!(*value.extract_map().get(&0).unwrap(), 1.0);
258	/// # 	Ok(())
259	/// # }
260	/// ```
261	pub fn new_kv(keys: Tensor<K>, values: Tensor<V>) -> Result<Self> {
262		let mut value_ptr = ptr::null_mut();
263		let values: [DynValue; 2] = [keys.into_dyn(), values.into_dyn()];
264		let value_ptrs: Vec<*const ort_sys::OrtValue> = values.iter().map(|c| c.ptr()).collect();
265		ortsys![
266			unsafe CreateValue(value_ptrs.as_ptr(), 2, ort_sys::ONNXType::ONNX_TYPE_MAP, &mut value_ptr)?;
267			nonNull(value_ptr)
268		];
269		Ok(Value {
270			inner: ValueInner::new_backed(
271				value_ptr,
272				ValueType::Map {
273					key: K::into_tensor_element_type(),
274					value: V::into_tensor_element_type()
275				},
276				None,
277				true,
278				Box::new(values)
279			),
280			_markers: PhantomData
281		})
282	}
