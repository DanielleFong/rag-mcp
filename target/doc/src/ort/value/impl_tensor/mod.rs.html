82impl DynTensor {
83	/// Construct a tensor via a given allocator with a given shape and datatype. The data in the tensor will be
84	/// **uninitialized**.
85	///
86	/// This can be used to create a tensor with data on a certain device. For example, to create a tensor with pinned
87	/// (CPU) memory for use with CUDA:
88	/// ```no_run
89	/// # use ort::{memory::{Allocator, MemoryInfo, MemoryType, AllocationDevice, AllocatorType}, session::Session, tensor::TensorElementType, value::DynTensor};
90	/// # fn main() -> ort::Result<()> {
91	/// # let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
92	/// let allocator = Allocator::new(
93	/// 	&session,
94	/// 	MemoryInfo::new(AllocationDevice::CUDA_PINNED, 0, AllocatorType::Device, MemoryType::CPUInput)?
95	/// )?;
96	///
97	/// let mut img_input = DynTensor::new(&allocator, TensorElementType::Float32, [1_usize, 128, 128, 3])?;
98	/// # Ok(())
99	/// # }
100	/// ```
101	pub fn new(allocator: &Allocator, data_type: TensorElementType, shape: impl Into<Shape>) -> Result<DynTensor> {
102		Self::new_inner(allocator, data_type, shape.into())
103	}
104
105	fn new_inner(allocator: &Allocator, data_type: TensorElementType, shape: Shape) -> Result<DynTensor> {
106		let mut value_ptr: *mut ort_sys::OrtValue = ptr::null_mut();
107
108		let shape_ptr: *const i64 = shape.as_ptr();
109		let shape_len = shape.len();
110
111		ortsys![
112			unsafe CreateTensorAsOrtValue(
113				allocator.ptr().cast_mut(),
114				shape_ptr,
115				shape_len,
116				data_type.into(),
117				&mut value_ptr
118			)?;
119			nonNull(value_ptr)
120		];
121
122		// `CreateTensorAsOrtValue` actually does not guarantee that the data allocated is zero'd out, so if we can, we should
123		// do it manually.
124		let memory_info = unsafe { MemoryInfo::from_value(value_ptr) }.expect("CreateTensorAsOrtValue returned non-tensor");
125		#[cfg(not(target_arch = "wasm32"))] // Data is not available at this point with `ort-web`, so this would just instantly fail
126		if memory_info.is_cpu_accessible() && data_type != TensorElementType::String {
127			let mut buffer_ptr: *mut ort_sys::c_void = ptr::null_mut();
128			ortsys![unsafe GetTensorMutableData(value_ptr.as_ptr(), &mut buffer_ptr)?];
129			if !buffer_ptr.is_null() {
130				unsafe { buffer_ptr.write_bytes(0, data_type.byte_size(shape.num_elements())) };
131			}
132		}
133
134		Ok(Value {
135			inner: ValueInner::new(
136				value_ptr,
137				ValueType::Tensor {
138					ty: data_type,
139					shape,
140					dimension_symbols: SymbolicDimensions::empty(shape_len)
141				},
142				Some(memory_info),
143				true
144			),
145			_markers: PhantomData
146		})
147	}
148}

150impl<Type: DefiniteTensorValueTypeMarker + ?Sized> Value<Type> {
151	/// Returns a mutable pointer to the tensor's data. The pointer may be null in the case of zero-sized tensors.
152	///
153	/// It's important to note that the resulting pointer may not point to CPU-accessible memory. In the case of a
154	/// tensor created on a different EP device, e.g. via [`Tensor::new`], the pointer returned by this function may be
155	/// a CUDA pointer, which would require a separate crate (like [`cudarc`](https://crates.io/crates/cudarc)) to access.
156	/// Use [`Tensor::memory_info`] & [`MemoryInfo::allocation_device`] to check which device the data resides on before
157	/// accessing it.
158	///
159	/// ```
160	/// # use ort::value::Tensor;
161	/// # fn main() -> ort::Result<()> {
162	/// let mut tensor = Tensor::<i64>::from_array((vec![5], vec![0, 1, 2, 3, 4]))?;
163	/// let ptr = tensor.data_ptr_mut().cast::<i64>();
164	/// unsafe {
165	/// 	*ptr.add(3) = 42;
166	/// };
167	///
168	/// let (_, extracted) = tensor.extract_tensor();
169	/// assert_eq!(&extracted, &[0, 1, 2, 42, 4]);
170	/// # Ok(())
171	/// # }
172	/// ```
173	pub fn data_ptr_mut(&mut self) -> *mut ort_sys::c_void {
174		let mut buffer_ptr: *mut ort_sys::c_void = ptr::null_mut();
175		ortsys![unsafe GetTensorMutableData(self.ptr_mut(), &mut buffer_ptr).expect("failed to get tensor data")]; // infallible
176		buffer_ptr
177	}
178
179	/// Returns an immutable pointer to the tensor's underlying data. The pointer may be null in the case of zero-sized
180	/// tensors.
181	///
182	/// It's important to note that the resulting pointer may not point to CPU-accessible memory. In the case of a
183	/// tensor created on a different EP device, e.g. via [`Tensor::new`], the pointer returned by this function may be
184	/// a CUDA pointer, which would require a separate crate (like [`cudarc`](https://crates.io/crates/cudarc)) to access.
185	/// Use [`Tensor::memory_info`] & [`MemoryInfo::allocation_device`] to check which device the data resides on before
186	/// accessing it.
187	///
188	/// ```
189	/// # use ort::value::Tensor;
190	/// # fn main() -> ort::Result<()> {
191	/// let tensor = Tensor::<i64>::from_array((vec![5], vec![0, 1, 2, 3, 4]))?;
192	/// let ptr = tensor.data_ptr().cast::<i64>();
193	/// assert_eq!(unsafe { *ptr.add(3) }, 3);
194	/// # Ok(())
195	/// # }
196	/// ```
197	pub fn data_ptr(&self) -> *const ort_sys::c_void {
198		let mut buffer_ptr: *mut ort_sys::c_void = ptr::null_mut();
199		ortsys![unsafe GetTensorMutableData(self.ptr().cast_mut(), &mut buffer_ptr).expect("failed to get tensor data")]; // infallible
200		buffer_ptr
201	}
202
203	/// Returns information about the device this tensor is allocated on.
204	///
205	/// ```
206	/// # use ort::{memory::{Allocator, AllocatorType, AllocationDevice, MemoryInfo, MemoryType}, session::Session, value::Tensor};
207	/// # fn main() -> ort::Result<()> {
208	/// let tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
209	/// // Tensors are allocated on CPU by default.
210	/// assert_eq!(tensor.memory_info().allocation_device(), AllocationDevice::CPU);
211	///
212	/// # if false {
213	/// # let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
214	/// let cuda_allocator = Allocator::new(
215	/// 	&session,
216	/// 	MemoryInfo::new(AllocationDevice::CUDA, 0, AllocatorType::Device, MemoryType::Default)?
217	/// )?;
218	/// let tensor = Tensor::<f32>::new(&cuda_allocator, [1_usize, 3, 224, 224])?;
219	/// assert_eq!(tensor.memory_info().allocation_device(), AllocationDevice::CUDA);
220	/// # }
221	/// # Ok(())
222	/// # }
223	/// ```
224	pub fn memory_info(&self) -> &MemoryInfo {
225		unsafe { self.inner.memory_info.as_ref().unwrap_unchecked() }
226	}
227}

229impl<T: IntoTensorElementType + Debug> Tensor<T> {
230	/// Converts from a strongly-typed [`Tensor<T>`] to a type-erased [`DynTensor`].
231	///
232	/// ```
233	/// # use ort::{memory::Allocator, value::Tensor};
234	/// # fn main() -> ort::Result<()> {
235	/// let tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
236	/// let tensor_dyn = tensor.upcast();
237	/// assert!(tensor_dyn.try_extract_tensor::<f32>().is_ok());
238	/// assert!(tensor_dyn.try_extract_tensor::<i64>().is_err());
239	/// # Ok(())
240	/// # }
241	/// ```
242	#[inline]
243	pub fn upcast(self) -> DynTensor {
244		unsafe { self.transmute_type() }
245	}
246
247	/// Creates a type-erased [`DynTensorRef`] from a strongly-typed [`Tensor<T>`].
248	///
249	/// ```
250	/// # use ort::{memory::Allocator, value::Tensor};
251	/// # fn main() -> ort::Result<()> {
252	/// let tensor = Tensor::<f32>::new(&Allocator::default(), [1_usize, 3, 224, 224])?;
253	/// let tensor_dyn = tensor.upcast_ref();
254	///
255	/// let (_, original_extract) = tensor.extract_tensor();
256	/// let (_, ref_extract) = tensor_dyn.try_extract_tensor::<f32>()?;
257	/// assert_eq!(original_extract, ref_extract);
258	/// # Ok(())
259	/// # }
260	/// ```
261	#[inline]
262	pub fn upcast_ref(&self) -> DynTensorRef<'_> {
263		DynTensorRef::new(Value {
264			inner: Arc::clone(&self.inner),
265			_markers: PhantomData
266		})
267	}
268
269	/// Converts from a strongly-typed [`Tensor<T>`] to a mutable reference to a type-erased [`DynTensor`].
270	///
271	/// ```
272	/// # use ort::value::Tensor;
273	/// # fn main() -> ort::Result<()> {
274	/// let mut tensor = Tensor::<i64>::from_array((vec![5], vec![1, 2, 3, 4, 5]))?;
275	/// let mut tensor_dyn = tensor.upcast_mut();
276	///
277	/// let (_, mut_view) = tensor_dyn.try_extract_tensor_mut::<i64>()?;
278	/// mut_view[3] = 0;
279	///
280	/// let (_, original_view) = tensor.extract_tensor();
281	/// assert_eq!(original_view, &[1, 2, 3, 0, 5]);
282	/// # Ok(())
283	/// # }
284	/// ```
285	#[inline]
286	pub fn upcast_mut(&mut self) -> DynTensorRefMut<'_> {
287		DynTensorRefMut::new(Value {
288			inner: Arc::clone(&self.inner),
289			_markers: PhantomData
290		})
291	}

316impl<T: IntoTensorElementType + Clone + Debug, const N: usize> Index<[i64; N]> for Tensor<T> {
317	type Output = T;
318	fn index(&self, index: [i64; N]) -> &Self::Output {
319		// Interestingly, the `TensorAt` API doesn't check if the tensor is on CPU, so we have to perform the check ourselves.
320		if !self.memory_info().is_cpu_accessible() {
321			panic!("Cannot directly index a tensor which is not allocated on the CPU.");
322		}
323
324		let mut out: *mut ort_sys::c_void = ptr::null_mut();
325		ortsys![unsafe TensorAt(self.ptr().cast_mut(), index.as_ptr(), N, &mut out).expect("Failed to index tensor")];
326		unsafe { &*out.cast::<T>() }
327	}
328}
329impl<T: IntoTensorElementType + Clone + Debug, const N: usize> IndexMut<[i64; N]> for Tensor<T> {
330	fn index_mut(&mut self, index: [i64; N]) -> &mut Self::Output {
331		if !self.memory_info().is_cpu_accessible() {
332			panic!("Cannot directly index a tensor which is not allocated on the CPU.");
333		}
334
335		let mut out: *mut ort_sys::c_void = ptr::null_mut();
336		ortsys![unsafe TensorAt(self.ptr_mut(), index.as_ptr(), N, &mut out).expect("Failed to index tensor")];
337		unsafe { &mut *out.cast::<T>() }
338	}
339}
