19/// The type of a [`Value`][super::Value], or a session input/output.
20///
21/// ```
22/// # use std::sync::Arc;
23/// # use ort::{session::Session, tensor::{Shape, SymbolicDimensions}, value::{ValueType, Tensor}, tensor::TensorElementType};
24/// # fn main() -> ort::Result<()> {
25/// # 	let session = Session::builder()?.commit_from_file("tests/data/upsample.onnx")?;
26/// // `ValueType`s can be obtained from session inputs/outputs:
27/// let input = &session.inputs()[0];
28/// assert_eq!(input.dtype(), &ValueType::Tensor {
29/// 	ty: TensorElementType::Float32,
30/// 	// Our model's input has 3 dynamic dimensions, represented by -1
31/// 	shape: Shape::new([-1, -1, -1, 3]),
32/// 	// Dynamic dimensions may also have names.
33/// 	dimension_symbols: SymbolicDimensions::new([
34/// 		"unk__31".to_string(),
35/// 		"unk__32".to_string(),
36/// 		"unk__33".to_string(),
37/// 		String::default()
38/// 	])
39/// });
40///
41/// // ...or by `Value`s created in Rust or output by a session.
42/// let value = Tensor::from_array(([5usize], vec![1_i64, 2, 3, 4, 5].into_boxed_slice()))?;
43/// assert_eq!(value.dtype(), &ValueType::Tensor {
44/// 	ty: TensorElementType::Int64,
45/// 	shape: Shape::new([5]),
46/// 	dimension_symbols: SymbolicDimensions::new([String::default()])
47/// });
48/// # 	Ok(())
49/// # }
50/// ```
51#[derive(Debug, PartialEq, Eq, Clone)]
52pub enum ValueType {
53	/// Value is a tensor/multi-dimensional array.
54	Tensor {
55		/// Element type of the tensor.
56		ty: TensorElementType,
57		/// Shape of the tensor. If an exact dimension is not known (i.e. a dynamic dimension as part of an
58		/// [`Outlet`]), the dimension will be `-1`.
59		///
60		/// Actual tensor values (i.e. not [`Outlet`] definitions), which have a known dimension, will
61		/// always have non-negative dimensions.
62		shape: Shape,
63		dimension_symbols: SymbolicDimensions
64	},
65	/// A sequence (vector) of other `Value`s.
66	///
67	/// [Per ONNX spec](https://onnx.ai/onnx/intro/concepts.html#other-types), only sequences of tensors and maps are allowed.
68	Sequence(Box<ValueType>),
69	/// A map/dictionary from one element type to another.
70	Map {
71		/// The map key type. Allowed types are:
72		/// - [`TensorElementType::Int8`]
73		/// - [`TensorElementType::Int16`]
74		/// - [`TensorElementType::Int32`]
75		/// - [`TensorElementType::Int64`]
76		/// - [`TensorElementType::Uint8`]
77		/// - [`TensorElementType::Uint16`]
78		/// - [`TensorElementType::Uint32`]
79		/// - [`TensorElementType::Uint64`]
80		/// - [`TensorElementType::String`]
81		key: TensorElementType,
82		/// The map value type.
83		value: TensorElementType
84	},
85	/// An optional value, which may or may not contain a [`Value`][super::Value].
86	Optional(Box<ValueType>)
87}

90	pub(crate) unsafe fn from_type_info(typeinfo_ptr: NonNull<ort_sys::OrtTypeInfo>) -> Self {
91		let _guard = util::run_on_drop(|| {
92			ortsys![unsafe ReleaseTypeInfo(typeinfo_ptr.as_ptr())];
93		});
94
95		let mut ty: ort_sys::ONNXType = ort_sys::ONNXType::ONNX_TYPE_UNKNOWN;
96		ortsys![unsafe GetOnnxTypeFromTypeInfo(typeinfo_ptr.as_ptr(), &mut ty).expect("infallible")];
97		match ty {
98			ort_sys::ONNXType::ONNX_TYPE_TENSOR | ort_sys::ONNXType::ONNX_TYPE_SPARSETENSOR => {
99				let mut info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
100				ortsys![unsafe CastTypeInfoToTensorInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
101				unsafe { extract_data_type_from_tensor_info(info_ptr) }
102			}
103			ort_sys::ONNXType::ONNX_TYPE_SEQUENCE => {
104				let mut info_ptr: *const ort_sys::OrtSequenceTypeInfo = ptr::null_mut();
105				ortsys![unsafe CastTypeInfoToSequenceTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
106
107				let mut element_type_info: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
108				ortsys![unsafe GetSequenceElementType(info_ptr.as_ptr(), &mut element_type_info).expect("infallible"); nonNull(element_type_info)];
109				let _guard = util::run_on_drop(|| {
110					ortsys![unsafe ReleaseTypeInfo(element_type_info.as_ptr())];
111				});
112
113				let mut ty: ort_sys::ONNXType = ort_sys::ONNXType::ONNX_TYPE_UNKNOWN;
114				ortsys![unsafe GetOnnxTypeFromTypeInfo(element_type_info.as_ptr(), &mut ty).expect("infallible")];
115
116				match ty {
117					ort_sys::ONNXType::ONNX_TYPE_TENSOR => {
118						let mut info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
119						ortsys![unsafe CastTypeInfoToTensorInfo(element_type_info.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
120						let ty = unsafe { extract_data_type_from_tensor_info(info_ptr) };
121						ValueType::Sequence(Box::new(ty))
122					}
123					ort_sys::ONNXType::ONNX_TYPE_MAP => {
124						let mut info_ptr: *const ort_sys::OrtMapTypeInfo = ptr::null_mut();
125						ortsys![unsafe CastTypeInfoToMapTypeInfo(element_type_info.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
126						let ty = unsafe { extract_data_type_from_map_info(info_ptr) };
127						ValueType::Sequence(Box::new(ty))
128					}
129					_ => unreachable!()
130				}
131			}
132			ort_sys::ONNXType::ONNX_TYPE_MAP => {
133				let mut info_ptr: *const ort_sys::OrtMapTypeInfo = ptr::null_mut();
134				ortsys![unsafe CastTypeInfoToMapTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
135				unsafe { extract_data_type_from_map_info(info_ptr) }
136			}
137			ort_sys::ONNXType::ONNX_TYPE_OPTIONAL => {
138				let mut info_ptr: *const ort_sys::OrtOptionalTypeInfo = ptr::null_mut();
139				ortsys![unsafe CastTypeInfoToOptionalTypeInfo(typeinfo_ptr.as_ptr(), &mut info_ptr).expect("infallible"); nonNull(info_ptr)];
140
141				let mut contained_type: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
142				ortsys![unsafe GetOptionalContainedTypeInfo(info_ptr.as_ptr(), &mut contained_type).expect("infallible"); nonNull(contained_type)];
143
144				ValueType::Optional(Box::new(unsafe { ValueType::from_type_info(contained_type) }))
145			}
146			_ => unreachable!()
147		}
148	}

150	pub(crate) fn to_tensor_type_info(&self) -> Option<*mut ort_sys::OrtTensorTypeAndShapeInfo> {
151		match self {
152			Self::Tensor { ty, shape, dimension_symbols } => {
153				let mut info_ptr = ptr::null_mut();
154				ortsys![unsafe CreateTensorTypeAndShapeInfo(&mut info_ptr).expect("infallible")];
155				ortsys![unsafe SetTensorElementType(info_ptr, (*ty).into()).expect("infallible")];
156				ortsys![unsafe SetDimensions(info_ptr, shape.as_ptr(), shape.len()).expect("infallible")];
157				with_cstr_ptr_array(dimension_symbols, &|ptrs| {
158					ortsys![unsafe SetSymbolicDimensions(info_ptr, ptrs.as_ptr().cast_mut(), dimension_symbols.len()).expect("infallible")];
159					Ok(())
160				})
161				.expect("invalid dimension symbols");
162				Some(info_ptr)
163			}
164			_ => None
165		}
166	}

168	/// Converts this type to an [`ort_sys::OrtTypeInfo`] using the Model Editor API, so it shouldn't be used outside of
169	/// `crate::editor`
170	pub(crate) fn to_type_info(&self) -> Result<*mut ort_sys::OrtTypeInfo> {
171		let mut info_ptr: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
172		match self {
173			Self::Tensor { .. } => {
174				let tensor_type_info = self.to_tensor_type_info().expect("infallible");
175				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTensorTypeAndShapeInfo(tensor_type_info)]);
176				ortsys![@editor: unsafe CreateTensorTypeInfo(tensor_type_info, &mut info_ptr)?];
177			}
178			Self::Map { .. } => {
179				todo!();
180			}
181			Self::Sequence(ty) => {
182				let el_type = ty.to_type_info()?;
183				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(el_type)]);
184				ortsys![@editor: unsafe CreateSequenceTypeInfo(el_type, &mut info_ptr)?];
185			}
186			Self::Optional(ty) => {
187				let ty = ty.to_type_info()?;
188				let _guard = util::run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(ty)]);
189				ortsys![@editor: unsafe CreateOptionalTypeInfo(ty, &mut info_ptr)?];
190			}
191		}
192		Ok(info_ptr)
193	}

255impl fmt::Display for ValueType {
256	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
257		match self {
258			ValueType::Tensor { ty, shape, dimension_symbols } => {
259				write!(f, "Tensor<{ty}>(")?;
260				for (i, dimension) in shape.iter().copied().enumerate() {
261					if dimension == -1 {
262						let sym = &dimension_symbols[i];
263						if sym.is_empty() {
264							f.write_str("dyn")?;
265						} else {
266							f.write_str(sym)?;
267						}
268					} else {
269						dimension.fmt(f)?;
270					}
271					if i != shape.len() - 1 {
272						f.write_str(", ")?;
273					}
274				}
275				f.write_str(")")?;
276				Ok(())
277			}
278			ValueType::Map { key, value } => write!(f, "Map<{key}, {value}>"),
279			ValueType::Sequence(inner) => write!(f, "Sequence<{inner}>"),
280			ValueType::Optional(inner) => write!(f, "Option<{inner}>")
281		}
282	}

308	pub(crate) fn into_editor_value_info(self) -> Result<NonNull<ort_sys::OrtValueInfo>> {
309		let type_info = self.dtype.to_type_info()?;
310		let _guard = run_on_drop(|| ortsys![unsafe ReleaseTypeInfo(type_info)]);
311
312		let ptr = with_cstr(self.name.as_bytes(), &|name| {
313			let mut ptr: *mut ort_sys::OrtValueInfo = ptr::null_mut();
314			ortsys![@editor: unsafe CreateValueInfo(name.as_ptr(), type_info, &mut ptr)?; nonNull(ptr)];
315			Ok(ptr)
316		})?;
317		Ok(ptr)
318	}

321pub(crate) unsafe fn extract_data_type_from_tensor_info(info_ptr: NonNull<ort_sys::OrtTensorTypeAndShapeInfo>) -> ValueType {
322	let mut type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
323	ortsys![unsafe GetTensorElementType(info_ptr.as_ptr(), &mut type_sys).expect("infallible")];
324	assert_ne!(type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
325	// This transmute should be safe since its value is read from GetTensorElementType, which we must trust
326	let mut num_dims = 0;
327	ortsys![unsafe GetDimensionsCount(info_ptr.as_ptr(), &mut num_dims).expect("infallible")];
328
329	let mut node_dims = Shape::empty(num_dims);
330	ortsys![unsafe GetDimensions(info_ptr.as_ptr(), node_dims.as_mut_ptr(), num_dims).expect("infallible")];
331
332	let mut symbolic_dims: SmallVec<[_; 4]> = smallvec![ptr::null(); num_dims];
333	ortsys![unsafe GetSymbolicDimensions(info_ptr.as_ptr(), symbolic_dims.as_mut_ptr(), num_dims).expect("infallible")];
334
335	let dimension_symbols = symbolic_dims
336		.into_iter()
337		.map(|c| unsafe { CStr::from_ptr(c) }.to_str().map_or_else(|_| String::new(), str::to_string))
338		.collect();
339
340	ValueType::Tensor {
341		ty: type_sys.into(),
342		shape: node_dims,
343		dimension_symbols
344	}
345}

347unsafe fn extract_data_type_from_map_info(info_ptr: NonNull<ort_sys::OrtMapTypeInfo>) -> ValueType {
348	let mut key_type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
349	ortsys![unsafe GetMapKeyType(info_ptr.as_ptr(), &mut key_type_sys).expect("infallible")];
350	assert_ne!(key_type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
351
352	let mut value_type_info: *mut ort_sys::OrtTypeInfo = ptr::null_mut();
353	ortsys![unsafe GetMapValueType(info_ptr.as_ptr(), &mut value_type_info).expect("infallible")];
354	let _guard = util::run_on_drop(|| {
355		ortsys![unsafe ReleaseTypeInfo(value_type_info)];
356	});
357
358	let mut value_info_ptr: *const ort_sys::OrtTensorTypeAndShapeInfo = ptr::null_mut();
359	ortsys![unsafe CastTypeInfoToTensorInfo(value_type_info, &mut value_info_ptr).expect("infallible")];
360	let mut value_type_sys = ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED;
361	ortsys![unsafe GetTensorElementType(value_info_ptr, &mut value_type_sys).expect("infallible")];
362	assert_ne!(value_type_sys, ort_sys::ONNXTensorElementDataType::ONNX_TENSOR_ELEMENT_DATA_TYPE_UNDEFINED);
363
364	ValueType::Map {
365		key: key_type_sys.into(),
366		value: value_type_sys.into()
367	}
368}
