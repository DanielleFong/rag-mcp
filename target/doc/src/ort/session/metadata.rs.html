20/// Container for model metadata, including name & producer information.
21pub struct ModelMetadata<'s> {
22	ptr: NonNull<ort_sys::OrtModelMetadata>,
23	allocator: Allocator,
24	_p: PhantomData<&'s ()>
25}

29		crate::logging::create!(ModelMetadata, ptr);

37	/// Gets the model description, returning an error if no description is present.
38	pub fn description(&self) -> Option<String> {
39		let mut str_bytes: *mut c_char = ptr::null_mut();
40		ortsys![@ort: unsafe ModelMetadataGetDescription(self.ptr.as_ptr(), self.allocator.ptr().cast_mut(), &mut str_bytes) as Result].ok()?;
41		unsafe { AllocatedString::from_ptr(str_bytes, &self.allocator) }
42			.ok()

89	/// Fetch the value of a custom metadata key. Returns `Ok(None)` if the key is not found.
90	pub fn custom(&self, key: &str) -> Option<String> {
91		let str_bytes = with_cstr(key.as_bytes(), &|key| {
92			let mut str_bytes: *mut c_char = ptr::null_mut();
93			ortsys![unsafe ModelMetadataLookupCustomMetadataMap(self.ptr.as_ptr(), self.allocator.ptr().cast_mut(), key.as_ptr(), &mut str_bytes)?];
94			Ok(str_bytes)
95		})
96		.ok()?;
97		unsafe { AllocatedString::from_ptr(str_bytes, &self.allocator) }
98			.ok()
99			.map(|s| s.to_string())
100	}
101
102	pub fn custom_keys(&self) -> Result<Vec<String>> {
103		let mut keys: *mut *mut c_char = ptr::null_mut();
104		let mut key_len = 0;
105		ortsys![unsafe ModelMetadataGetCustomMetadataMapKeys(self.ptr.as_ptr(), self.allocator.ptr().cast_mut(), &mut keys, &mut key_len)?];
106		if key_len != 0 && !keys.is_null() {
107			let res = unsafe { slice::from_raw_parts(keys, key_len as usize) }
108				.iter()
109				.map(|c| {
110					let res = char_p_to_string(*c);
111					unsafe { self.allocator.free(*c) };
112					res
113				})
114				.collect();
115			unsafe { self.allocator.free(keys) };
116			res
117		} else {
118			Ok(Vec::new())
119		}
120	}
121}
