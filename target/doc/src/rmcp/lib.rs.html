3//! The official Rust SDK for the Model Context Protocol (MCP).
4//!
5//! The MCP is a protocol that allows AI assistants to communicate with other
6//! services. `rmcp` is the official Rust implementation of this protocol.
7//!
8//! There are two ways in which the library can be used, namely to build a
9//! server or to build a client.
10//!
11//! ## Server
12//!
13//! A server is a service that exposes capabilities. For example, a common
14//! use-case is for the server to make multiple tools available to clients such
15//! as Claude Desktop or the Cursor IDE.
16//!
17//! For example, to implement a server that has a tool that can count, you would
18//! make an object for that tool and add an implementation with the `#[tool_router]` macro:
19//!
20//! ```rust
21//! use std::sync::Arc;
22//! use rmcp::{ErrorData as McpError, model::*, tool, tool_router, handler::server::tool::ToolRouter};
23//! use tokio::sync::Mutex;
24//!
25//! #[derive(Clone)]
26//! pub struct Counter {
27//!     counter: Arc<Mutex<i32>>,
28//!     tool_router: ToolRouter<Self>,
29//! }
30//!
31//! #[tool_router]
32//! impl Counter {
33//!     fn new() -> Self {
34//!         Self {
35//!             counter: Arc::new(Mutex::new(0)),
36//!             tool_router: Self::tool_router(),
37//!         }
38//!     }
39//!
40//!     #[tool(description = "Increment the counter by 1")]
41//!     async fn increment(&self) -> Result<CallToolResult, McpError> {
42//!         let mut counter = self.counter.lock().await;
43//!         *counter += 1;
44//!         Ok(CallToolResult::success(vec![Content::text(
45//!             counter.to_string(),
46//!         )]))
47//!     }
48//! }
49//! ```
50//!
51//! ### Structured Output
52//!
53//! Tools can also return structured JSON data with schemas. Use the [`Json`] wrapper:
54//!
55//! ```rust
56//! # use rmcp::{tool, tool_router, handler::server::{tool::ToolRouter, wrapper::Parameters}, Json};
57//! # use schemars::JsonSchema;
58//! # use serde::{Serialize, Deserialize};
59//! #
60//! #[derive(Serialize, Deserialize, JsonSchema)]
61//! struct CalculationRequest {
62//!     a: i32,
63//!     b: i32,
64//!     operation: String,
65//! }
66//!
67//! #[derive(Serialize, Deserialize, JsonSchema)]
68//! struct CalculationResult {
69//!     result: i32,
70//!     operation: String,
71//! }
72//!
73//! # #[derive(Clone)]
74//! # struct Calculator {
75//! #     tool_router: ToolRouter<Self>,
76//! # }
77//! #
78//! # #[tool_router]
79//! # impl Calculator {
80//! #[tool(name = "calculate", description = "Perform a calculation")]
81//! async fn calculate(&self, params: Parameters<CalculationRequest>) -> Result<Json<CalculationResult>, String> {
82//!     let result = match params.0.operation.as_str() {
83//!         "add" => params.0.a + params.0.b,
84//!         "multiply" => params.0.a * params.0.b,
85//!         _ => return Err("Unknown operation".to_string()),
86//!     };
87//!     
88//!     Ok(Json(CalculationResult { result, operation: params.0.operation }))
89//! }
90//! # }
91//! ```
92//!
93//! The `#[tool]` macro automatically generates an output schema from the `CalculationResult` type.
94//!
95//! Next also implement [ServerHandler] for your server type and start the server inside
96//! `main` by calling `.serve(...)`. See the examples directory in the repository for more information.
97//!
98//! ## Client
99//!
100//! A client can be used to interact with a server. Clients can be used to get a
101//! list of the available tools and to call them. For example, we can `uv` to
102//! start a MCP server in Python and then list the tools and call `git status`
103//! as follows:
104//!
105//! ```rust
106//! use anyhow::Result;
107//! use rmcp::{model::CallToolRequestParam, service::ServiceExt};
108//! #[cfg(feature = "transport-child-process")]
109//! #[cfg_attr(docsrs, doc(cfg(feature = "transport-child-process")))]
110//! use rmcp::transport::{TokioChildProcess, ConfigureCommandExt};
111//! use tokio::process::Command;
112//!
113//! #[cfg(feature = "transport-child-process")]
114//! #[cfg_attr(docsrs, doc(cfg(feature = "transport-child-process")))]
115//! async fn client() -> Result<()> {
116//!     let service = ().serve(TokioChildProcess::new(Command::new("uvx").configure(|cmd| {
117//!         cmd.arg("mcp-server-git");
118//!     }))?).await?;
119//!
120//!     // Initialize
121//!     let server_info = service.peer_info();
122//!     println!("Connected to server: {server_info:#?}");
123//!
124//!     // List tools
125//!     let tools = service.list_tools(Default::default()).await?;
126//!     println!("Available tools: {tools:#?}");
127//!
128//!     // Call tool 'git_status' with arguments = {"repo_path": "."}
129//!     let tool_result = service
130//!         .call_tool(CallToolRequestParam {
131//!             name: "git_status".into(),
132//!             arguments: serde_json::json!({ "repo_path": "." }).as_object().cloned(),
133//!             task: None,
134//!         })
135//!         .await?;
136//!     println!("Tool result: {tool_result:#?}");
137//!
138//!     service.cancel().await?;
139//!     Ok(())
140//! }
141//! ```
